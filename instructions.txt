BACKUP DATABASE TestDB  
TO DISK = '/var/opt/mssql/data/TestDB_Backup.bak'  
WITH FORMAT, INIT, COMPRESSION, STATS = 10;

FROM DISK = '/var/opt/mssql/data/TestDB_Backup.bak'

MOVE 'TestDB_log' TO '/var/opt/mssql/data/TestDB.ldf',
